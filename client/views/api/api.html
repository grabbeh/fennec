API madness coming soon...
  • // General portfolio search function based on query parameters "group" and "country"
    GET '/api/searchTrademarks/:portfolio'
    e.g. '/api/searchTrademarks/ACME INC?group=ACME INC&country=USA'
  • // Country data (ISO, coordinates etc) - optionally accepts query parameter 'portfolio' in which case returns // country data limited to countries where there are trade marks in that portfolio
    GET '/api/countryData'
    e.g. '/api/countryData?portfolio=ACME INC'
  • // Provide geojson for group of marks
    GET '/api/world/:portfolio'
  • // Filtered world on basis of given array of marks
    POST '/api/world/:portfolio'
  • // Country-limited trade marks
    // Gets a list of all trade marks for a given portfolio in a given country (ISO code)
    GET '/api/country/:portfolio/:country'
  • // In addition to the above, accepts array of trade marks and provides only those marks in a portfolio
    POST '/api/country/:portfolio/:country'
  • // Get group of marks
    GET '/api/trademarks/:portfolio'
  • // Edit individual trade marks
    GET '/api/trademark/:id'
    POST '/api/trademark'
    PUT '/api/trademark/:id'
    DELETE '/api/trademark/:id'
  • // Returns basic list of marks for portfolio - accepts optional query for ISO code in // which case returns list of marks for given country
    GET '/api/list/:portfolio'
  • // Edit group
    POST '/api/editGroup/:portfolio/:mark'
    POST '/api/addLogoToGroup/:portfolio/:mark'
    POST '/api/editMarksInCountry/:portfolio'
  • // Expiry dates
    // accepts query for group in addition to portfolio
    GET '/api/expirydates/:portfolio'
    GET '/api/expirydates/:portfolio/:country'
    GET '/api/expiriesForYear/:portfolio'
  • // Activities
    GET '/api/activities/:portfolio'
